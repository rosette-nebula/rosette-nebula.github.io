类型判断

js 对变量类型判断的一些方法。

typeof

// 函数
function demo() {
  console.log('这是案例');
}

// 类
class Person {
  name = '张三';
}

const p = new Person();

console.log('Number : ', typeof 123); // number
console.log('NaN 属于数字 : ', typeof NaN); // number
console.log('BigInt : ', typeof 123n); // bigint
console.log('String : ', typeof '字符串'); // string
console.log('Boolean : ', typeof true); // boolean
console.log('Null : ', typeof null); // object
console.log('Undefined : ', typeof undefined); // undefined
console.log('Function : ', typeof demo); // function
console.log('Array : ', typeof []); // object
console.log('Object : ', typeof {}); // object
console.log('Person : ', typeof Person); // function
console.log('p : ', typeof p); // object

instance

语法 object instance constructor instanceof 运算符用来检测 constructor.prototype 是否存在于参数 object 的原型链上。

function Car(make, model, year) {
  this.make = make;
  this.model = model;
  this.year = year;
}
const auto = new Car('Honda', 'Accord', 1998);

console.log(auto instanceof Car); // true

console.log(auto instanceof Object); // true

Object.prototype.toString.call

// 函数
function demo() {
  console.log('这是案例');
}

// 类
class Person {
  name = '张三';
}

const p = new Person();

console.log('Number : ', Object.prototype.toString.call(123)); // [object Number]
console.log('NaN 属于数字 : ', Object.prototype.toString.call(NaN)); // [object Number]
console.log('BigInt : ', Object.prototype.toString.call(123n)); // [object BigInt]
console.log('String : ', Object.prototype.toString.call('字符串')); // [object String]
console.log('Boolean : ', Object.prototype.toString.call(true)); // [object Boolean]
console.log('Null : ', Object.prototype.toString.call(null)); // [object Null]
console.log('Undefined : ', Object.prototype.toString.call(undefined)); // [object Undefined]
console.log('Function : ', Object.prototype.toString.call(demo)); // [object Function]
console.log('Array : ', Object.prototype.toString.call([])); // [object Array]
console.log('Object : ', Object.prototype.toString.call({})); // [object Object]
console.log('Person : ', Object.prototype.toString.call(Person)); // [object Function]
console.log('p : ', Object.prototype.toString.call(p)); // [object Object]

Object.prototype.toString.call 原理

  • 所有数据类型都是对象的一种类型，而Object.prototype.toString可以返回当前调用者的对象类型。

  • 为什么Object.prototype.toString 要加call? 为什么不用 变量本身原型对象上的 toString

    • 为了改变toString调用是的this指向，要判断那种类型的变量，就指向那个变量。
    • Object.prototype.toString 会返回当前 this 的类型。而其他类型的 toString 方法是重写了Object的toString方法。如以下列子:
console.log(Object.prototype.toString()); // [object Object]
console.log(Array.prototype.toString()); // 空字符串

// 删除Array 原型对象上的 toString 方法
delete Array.prototype.toString;
// 通过原型链 调用 Object 原型对象上的 toString 方法
console.log(Array.prototype.toString()); // [object Array]

封装 Object.prototype.toString.call 判断所有基本类型

/**
 * 获取变量的基本类型
 * 1. 通过Object.prototype.toString.call 获取变量的类型
 * 2. 通过正则表达式去除类型的多余字符串
 * 3. 将类型全部转为小写
 * @param {*} variable - 需要被获取类型的变量
 * @returns {String}
 */
function getType(variable) {
  return Object.prototype.toString
    .call(variable)
    .replace(/\[object (.*)\]/g, '$1')
    .toLowerCase();
}
console.log(getType(123)); // number

如何判断一个变量是否为 NaN

  • 通过typeof或者Object.prototype.toString.call都获取到数字和NaN的类型为 number.
  • 通过Number.isNaN 方法判断
const a = NaN
const b = 123

console.log(Number.isNaN(a)) // true
console.log(Number.isNaN(b)) // false
